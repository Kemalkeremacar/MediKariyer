# MediKariyer Backend Environment Variables
# Kopyalayın: Backend klasörüne .env dosyası oluşturup bu değerleri kendi bilgilerinizle doldurun

# ==================== VERİTABANI AYARLARI ====================
# SQL Server bağlantı bilgileri
# ÖNEMLİ: SQL Server Browser ve TCP/IP'yi enable etmelisiniz!
# SQL Server Configuration Manager'da:
# 1. TCP/IP -> Enable (Protocols for SQLEXPRESS altında)
# 2. SQL Server Browser servisi -> Start
# 3. SQL Server (SQLEXPRESS) -> Restart

DB_HOST=localhost
DB_INSTANCE=SQLEXPRESS
DB_PORT=1433
DB_NAME=MEDIKARIYER
DB_USER=sa
DB_PASSWORD=YourSQLPassword123!
DB_ENCRYPT=false
DB_TRUST_SERVER_CERTIFICATE=true

# ==================== JWT AYARLARI ====================
# Token üretimi için gizli anahtarlar (GÜÇLÜ RANDOM STRİNGLER OLUŞTURUN!)
